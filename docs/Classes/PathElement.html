PathElement

public final class PathElement : CustomStringConvertible

Represents a path inside of a JSON file. This path can then be used later to refer to that location using only a name and not the full path

  //The first element (or root) in the path
  let rootPath = PathElement(name: "response")
  //The rest of the path is now the first element in the responses array, followed by a response
  //node followed by an ob node
  rootPath.withChild("responses", childAtIndex: 0)
   .withChild("response")
   .withChild("ob")
  //The name (or key) that can be used to refer to this path later
  let observation = "observation"
  //Assign this path to a JsonPath which combines the path with the name
  let observationPath = JsonPath(name: observation, rootPath: rootPath)
  //Instantiate a Json object giving the json data and paths (in this case, just 1)
  let json = try Json(jsonData: json, paths:observationPath)
  //Now a value can be retrived, using the string key and an element name
  let temperature:Int = try json.intValue(observation, key: tempKey)
